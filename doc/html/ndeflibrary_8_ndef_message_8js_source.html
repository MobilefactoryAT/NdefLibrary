107  NdefLibrary.NdefMessage.fromByteArray = function(message){
108 
109  var result = new NdefLibrary.NdefMessage();
110 
111  var seenMessageBegin = false;
112  var seenMessageEnd = false;
113 
114  var partialChunk = new Array();
115  var record = new NdefLibrary.NdefRecord();
116 
117  var i = 0;
118  while (i < message.length)
119  {
120  // console.log("Parsing byte[] to NDEF message. New record starts at {0}", i);
121 
122  // Parse flags out of NDEF message header
123  var messageBegin = (message[i] & 0x80) != 0;
124  var messageEnd = (message[i] & 0x40) != 0;
125  var cf = (message[i] & 0x20) != 0;
126  var sr = (message[i] & 0x10) != 0;
127  var il = (message[i] & 0x08) != 0;
128 
129  // var typeNameFormat = (NdefLibrary.NdefRecord.TypeNameFormatType)(message[i] & 0x07);
130  var typeNameFormat = (message[i] & 0x07);
131 
132  // console.log("TypeNameFormat: " + typeNameFormat+" message[i]:"+message[i]);
133  // console.log("ShortRecord: " + (sr ? "yes" : "no"));
134  // console.log("Id Length present: " + (il ? "yes" : "no"));
135 
136  if (messageBegin && seenMessageBegin)
137  {
138  throw "NdefExceptionMessages.ExMessageBeginLate";
139  }
140  else if (!messageBegin && !seenMessageBegin)
141  {
142  throw "NdefExceptionMessages.ExMessageBeginMissing";
143  }
144  else if (messageBegin && !seenMessageBegin)
145  {
146  seenMessageBegin = true;
147  }
148 
149  if (messageEnd && seenMessageEnd)
150  {
151  throw "NdefExceptionMessages.ExMessageEndLate";
152  }
153  else if (messageEnd && !seenMessageEnd)
154  {
155  seenMessageEnd = true;
156  }
157 
158  if (cf && (typeNameFormat != NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged) && partialChunk.length > 0)
159  {
160  throw "NdefExceptionMessages.ExMessagePartialChunk";
161  }
162 
163  // Header length
164  var headerLength = 1;
165  headerLength += (sr) ? 1 : 4;
166  headerLength += (il) ? 1 : 0;
167 
168  if (i + headerLength >= message.length)
169  {
170  throw "NdefExceptionMessages.ExMessageUnexpectedEnd";
171  }
172 
173  // Type length
174  var typeLength = message[++i];
175 
176  if ((typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged) && (typeLength != 0))
177  {
178  throw "NdefExceptionMessages.ExMessageInvalidChunkedType";
179  }
180 
181  // Payload length (short record?)
182  var payloadLength;
183  if (sr)
184  {
185  // Short record - payload length is a single octet
186  payloadLength = message[++i];
187  }
188  else
189  {
190  // No short record - payload length is four octets representing a 32 bit unsigned integer (MSB-first)
191  payloadLength = ((message[++i]) << 24);
192  payloadLength |= ((message[++i]) << 16);
193  payloadLength |= ((message[++i]) << 8);
194  payloadLength |= ((message[++i]) << 0);
195  }
196 
197  // ID length
198  var idLength;
199  idLength = (il ? message[++i] : 0);
200 
201  // Total length of content (= type + payload + ID)
202  var contentLength = typeLength + payloadLength + idLength;
203  if (i + contentLength >= message.length)
204  {
205  throw "NdefExceptionMessages.ExMessageUnexpectedEnd";
206  }
207 
208 
209  if ((typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged) && (idLength != 0))
210  {
211  throw "NdefExceptionMessages.ExMessageInvalidChunkedId";
212  }
213 
214  if (typeNameFormat != NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged)
215  {
216  record.setTypeNameFormat(typeNameFormat);
217  }
218 
219  // Read type
220  if (typeLength > 0)
221  {
222  // record.Type = new byte[typeLength];
223  // Array.Copy(message, (int)(++i), record.Type, 0, typeLength);
224 
225  var type = new Array();
226  arrayCopy(message,(++i), type, 0, typeLength);
227  record.setType(type);
228 
229  i += typeLength - 1;
230  }
231 
232  // Read ID
233  if (idLength > 0)
234  {
235  // record.Id = new byte[idLength];
236  // Array.Copy(message, (int)(++i), record.Id, 0, idLength);
237 
238  var id = new Array();
239  arrayCopy(message,(++i), id, 0, idLength);
240  record.setId(id);
241 
242  i += idLength - 1;
243  }
244 
245  // Read payload
246  if (payloadLength > 0)
247  {
248  // var payload = new byte[payloadLength];
249  // Array.Copy(message, (int)(++i), payload, 0, (int)payloadLength);
250 
251  var payload = new Array();
252  arrayCopy(message,(++i), payload, 0, payloadLength);
253  record.setPayload(payload);
254 
255  if (cf)
256  {
257  // chunked payload, except last
258  addToArray(partialChunk, payload);
259  // partialChunk.Write(payload, 0, payload.Length);
260  }
261  else if (typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged)
262  {
263  // last chunk of chunked payload
264  addToArray(partialChunk, payload);
265  record.setPayload(partialChunk);
266 
267  // partialChunk.Write(payload, 0, payload.Length);
268  // record.Payload = partialChunk.ToArray();
269  }
270  else
271  {
272  // non-chunked payload
273  record.setPayload(payload);
274  }
275 
276  i += payloadLength - 1;
277  }
278 
279  if (!cf)
280  {
281  // Add record to the message and create a new record for the next loop iteration
282  result.push(record);
283  record = new NdefLibrary.NdefRecord();
284  }
285 
286  if (!cf && seenMessageEnd)
287  break;
288 
289  // move to start of next record
290  ++i;
291  }
292 
293 
294  if (!seenMessageBegin && !seenMessageEnd)
295  {
296  throw "NdefExceptionMessages.ExMessageNoBeginOrEnd";
297  }
298 
299  return result;
300

309  ndefMessage.prototype.toByteArray = function(){
310 
311  var Count = this._records.length;
312 
313  // Empty message: single empty record
314  if (Count == 0)
315  {
316  var tmpNdefMessage = new NdefLibrary.NdefMessage(new NdefLibrary.NdefRecord());
317  return tmpNdefMessage.toByteArray();
318  }
319 
320  var m = new Array();
321 
322  for (var i = 0; i < Count; i++)
323  {
324  var record = this._records[i];
325 
326  var flags = record.getTypeNameFormat();
327 
328  // Message begin / end flags. If there is only one record in the message,
329  // both flags are set.
330  if (i == 0)
331  flags |= 0x80; // MB (message begin = first record in the message)
332  if (i == Count - 1)
333  flags |= 0x40; // ME (message end = last record in the message)
334 
335  // cf (chunked records) not supported yet
336 
337  // SR (Short Record)?
338  if (record.getPayload() == null || record.getPayload().length < 255)
339  flags |= 0x10;
340 
341  // ID present?
342  if (record.getId() != null && record.getId().length > 0)
343  flags |= 0x08;
344 
345  m.push(flags);
346 
347  // Type length
348  if (record.getType() != null) m.push(record.getType().length); else m.push(0);
349 
350  // Payload length 1 byte (SR) or 4 bytes
351  if (record.getPayload() == null)
352  m.push(0);
353  else
354  {
355  if ((flags & 0x10) != 0)
356  {
357  // SR
358  m.push(record.getPayload().length);
359  }
360  else
361  {
362  // No SR (Short Record)
363  var payloadLength = record.getPayload().length;
364 
365  m.push((payloadLength >> 24));
366  m.push((payloadLength >> 16));
367  m.push((payloadLength >> 8));
368  m.push((payloadLength & 0x000000ff));
369  }
370  }
371 
372  // ID length
373  if (record.getId() != null && (flags & 0x08) != 0)
374  m.push(record.getId().length);
375 
376  // Type length
377  if (record.getType() != null && record.getType().length > 0)
378  addToArray(m, record.getType());
379  // m.Write(record.Type, 0, record.Type.Length);
380 
381  // ID data
382  if (record.getId() != null && record.getId().length > 0)
383  addToArray(m, record.getId());
384  // m.Write(record.Id, 0, record.Id.Length);
385 
386  // Payload data
387  if (record.getPayload() != null && record.getPayload().length > 0)
388  addToArray(m, record.getPayload());
389  // m.Write(record.Payload, 0, record.Payload.Length);
390 
391  }
392 
393  return m;
394  };
