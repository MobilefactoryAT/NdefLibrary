244  ndefRecord.prototype.checkIfValid = function(){
245  // Check Type (according to TNF)
246  if (this._typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged ||
247  this._typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unknown){
248  // Unknown and unchanged TNF must have a type length of 0
249  if (!(this._type == null ||this._type.length == 0))
250  {
251  throw "NdefExceptionMessages.ExRecordUnchangedTypeName";
252  }
253  }
254  else{
255  // All other TNF (except Empty) should have a type set
256  // Proximity APIs unstable in some cases when no Type name is set
257  if (this._typeNameFormat != NdefLibrary.NdefRecord.TypeNameFormatType.Empty && (this._type == null || this._type.length == 0))
258  {
259  throw "NdefExceptionMessages.ExRecordNoType";

261  }

264  if (this._typeNameFormat == NdefLibrary.NdefRecord.TypeNameFormatType.Unchanged && !(this._id == null || this._id.length == 0))
